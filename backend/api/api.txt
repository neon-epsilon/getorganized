/hoursofwork/entries.php:
if GET:
  expect JSON payload:
    {
      limit: non-negative integer (default: 10)
      start: non-negative integer (default: 0)
    }
  if payload malformed:
    400, error message
  elif processing didn't work:
    500, error message
  else
    200, entries as JSON-list (id, date, category, amount)
elif POST:
  if payload malformed:
    400, error message
  elif processing didn't work:
    500, error message, optionally which POSTed fields where invalid
  else
    200, JSON-object with id of created row
elif DELETE:
  expect JSON-list of entry ids
  if payload malformed:
    400, error message
  elif processing didn't work:
    500, error message
  else
    200
else:
  405, error message
